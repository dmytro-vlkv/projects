// ── Tree text format: conversion ──


function dataToTreeText(node, indent = 0) {

const prefix = ' '.repeat(indent);

let line = prefix + node.name;

if (node.fullPath) line += ' | ' + node.fullPath;

if (node.desc) line += ' | ' + node.desc;

let lines = [line];

if (node.children) {

for (const child of node.children) {

lines = lines.concat(dataToTreeText(child, indent + 1));

}

}

return lines;

}


function treeTextToData(text) {

const lines = text.split('\n').filter(l => l.trim().length > 0);

if (lines.length === 0) throw new Error('Empty tree');


function getIndent(line) {

let count = 0;

for (let i = 0; i < line.length; i++) {

if (line[i] === ' ') count++;

else break;

}

return Math.floor(count / 2);

}


function parseLine(line) {

const trimmed = line.trim();

const parts = trimmed.split(' | ');

const name = parts[0].trim();

const fullPath = parts.length > 1 ? parts[1].trim() : '';

const desc = parts.length > 2 ? parts.slice(2).join(' | ').trim() : '';

const type = inferType(name, fullPath);

const node = { name, fullPath, desc, type };

return node;

}


function inferType(name, path) {

if (name.startsWith(':') || path.includes(':')) return 'param';

return 'page'; // will be upgraded later

}


// Build tree from indented lines

const rootNode = parseLine(lines[0]);

rootNode.type = 'root';

const stack = [{ node: rootNode, indent: getIndent(lines[0]) }];


for (let i = 1; i < lines.length; i++) {

const indent = getIndent(lines[i]);

const node = parseLine(lines[i]);


// Pop stack until we find the parent

while (stack.length > 1 && stack[stack.length - 1].indent >= indent) {

stack.pop();

}


const parent = stack[stack.length - 1].node;

if (!parent.children) parent.children = [];

parent.children.push(node);

stack.push({ node, indent });

}


// Second pass: upgrade types

function upgradeTypes(node, depth) {

if (depth <= 1 && node.children && node.children.length > 0 && node.type === 'page') {

// Check if it looks like a root (has a domain-like path)

if (node.fullPath && !node.fullPath.startsWith('/')) {

node.type = 'root';

}

}

if (node.children) {

for (const child of node.children) {

// If a non-param node has children and is at depth 2, it might be a surface

if (child.type === 'page' && child.children && child.children.length > 0 && depth >= 1) {

// Heuristic: top-level sections under an app root become surfaces

const parentIsRoot = node.type === 'root';

if (parentIsRoot) child.type = 'surface';

}

upgradeTypes(child, depth + 1);

}

}

}

upgradeTypes(rootNode, 0);


return rootNode;

}

// ── Editor ──


const editor = document.getElementById('tree-editor');

const statusEl = document.getElementById('editor-status');

let debounceTimer = null;


function populateEditor() {

if (currentFormat === 'tree') {

editor.value = dataToTreeText(currentData).join('\n');

} else {

editor.value = dataToJSON(currentData);

}

}


function switchFormat(fmt) {

// Save current edits first

try {

if (currentFormat === 'tree') {

currentData = treeTextToData(editor.value);

} else {

currentData = jsonToData(editor.value);

}

} catch(e) {

// If parsing fails, just keep current data

}


currentFormat = fmt;

document.querySelectorAll('.format-tab').forEach(t => t.classList.remove('active'));

document.querySelectorAll('.format-tab').forEach(t => {

if (t.textContent.toLowerCase() === fmt) t.classList.add('active');

});

populateEditor();

}


function showStatus(msg, isError) {

statusEl.textContent = msg;

statusEl.className = 'editor-status visible' + (isError ? ' error' : '');

setTimeout(() => { statusEl.className = 'editor-status'; }, 2500);

}


function applyChanges() {

try {

if (currentFormat === 'tree') {

currentData = treeTextToData(editor.value);

} else {

currentData = jsonToData(editor.value);

}

renderTree(currentData);

showStatus('Diagram updated', false);

} catch(e) {

showStatus('Parse error: ' + e.message, true);

}

}


function resetEditor() {

currentData = JSON.parse(JSON.stringify(initialData));

populateEditor();

renderTree(currentData);

showStatus('Reset to original', false);

}


// Auto-apply on edit (debounced)

editor.addEventListener('input', () => {

clearTimeout(debounceTimer);

debounceTimer = setTimeout(() => {

try {

let parsed;

if (currentFormat === 'tree') {

parsed = treeTextToData(editor.value);

} else {

parsed = jsonToData(editor.value);

}

currentData = parsed;

renderTree(currentData);

statusEl.textContent = '';

statusEl.className = 'editor-status';

} catch(e) {

statusEl.textContent = 'Parse error — fix syntax to update';

statusEl.className = 'editor-status visible error';

}

}, 600);

});

const tooltip = document.getElementById('tooltip');

const container = document.getElementById('chart-container');


function renderTree(data) {

// Clear

container.innerHTML = '';


const margin = { top: 30, right: 220, bottom: 30, left: 100 };

const nodeHeight = 44;


const root = d3.hierarchy(data);

const nodeCount = root.descendants().length;

const height = nodeCount * nodeHeight + margin.top + margin.bottom;

const width = Math.max(1500, container.clientWidth);


const svg = d3.select('#chart-container')

.append('svg')

.attr('width', width)

.attr('height', height);


const g = svg.append('g')

.attr('transform', `translate(${margin.left},${margin.top})`);


const treeLayout = d3.tree()

.size([height - margin.top - margin.bottom, width - margin.left - margin.right - 160])

.separation((a, b) => a.parent === b.parent ? 1 : 1.2);


treeLayout(root);


// Links

g.selectAll('.link')

.data(root.links())

.join('path')

.attr('class', 'link')

.attr('d', d3.linkHorizontal().x(d => d.y).y(d => d.x));


// Nodes

const node = g.selectAll('.node')

.data(root.descendants())

.join('g')

.attr('class', 'node')

.attr('transform', d => `translate(${d.y},${d.x})`);


function nodeColor(d) {

const t = d.data.type;

if (t === 'root') return '#B8420F';

if (t === 'surface') return '#1A1A18';

if (t === 'param') return 'transparent';

return 'rgba(26,26,24,.45)';

}


function nodeStroke(d) {

const t = d.data.type;

if (t === 'root') return '#B8420F';

if (t === 'surface') return '#1A1A18';

return 'rgba(26,26,24,.45)';

}


node.append('circle')

.attr('r', d => d.data.type === 'root' ? 7 : d.data.type === 'surface' ? 6 : 4.5)

.attr('fill', nodeColor)

.attr('stroke', nodeStroke)

.attr('stroke-dasharray', d => d.data.type === 'param' ? '3,2' : 'none');


node.append('text')

.attr('dy', -12).attr('x', 0).attr('text-anchor', 'middle')

.text(d => d.data.name)

.style('font-weight', d => (d.data.type === 'surface' || d.data.type === 'root') ? '600' : '400')

.style('font-size', d => d.data.type === 'root' ? '13px' : d.data.type === 'surface' ? '12px' : '11px');


node.append('text')

.attr('class', 'url-path').attr('dy', 18).attr('x', 0).attr('text-anchor', 'middle')

.text(d => d.data.fullPath);


// Tooltip

node.on('mouseenter', function(event, d) {

tooltip.innerHTML = `

<div class="tt-path">${d.data.fullPath || ''}</div>

<div>${d.data.desc || ''}</div>

`;

tooltip.style.opacity = 1;

tooltip.style.left = (event.pageX + 14) + 'px';

tooltip.style.top = (event.pageY - 14) + 'px';

});


node.on('mousemove', function(event) {

tooltip.style.left = (event.pageX + 14) + 'px';

tooltip.style.top = (event.pageY - 14) + 'px';

});


node.on('mouseleave', function() {

tooltip.style.opacity = 0;

});

}
